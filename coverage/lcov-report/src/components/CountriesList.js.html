export function countryToFlag(isoCode) {
	return isoCode
		? isoCode.toUpperCase().replace(/./g, char => String.fromCodePoint(char.charCodeAt(0) + 127397))
		: isoCode;
}

					{' '}
					({option.code}
					) +{option.phone}
				</>
			)}
			renderInput={params => (
				<TextField
					{...params}
					label='Residence'
					variant='outlined'
					size='small'
					fullWidth

						...params.inputProps,
						autoComplete: 'new-password', // disable autocomplete and autofill

				/>
			)}

		/>
	);
}
