export const disabledHandler = (props, state) => {
  let err = false;
  props.values.trips.forEach(trip => {
    if (trip.origin === '' || trip.destination === '' || trip.travelReasons === '') {
      err = true;
    }
  });
  if (Object.prototype.hasOwnProperty.call(props.errors, 'trips') || err || state.loading) {
    return true;
  }
  return false;
};
